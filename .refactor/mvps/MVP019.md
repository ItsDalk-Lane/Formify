```typescript
// src/service/action/loop/LoopActionService.ts
export class LoopActionService implements IActionService {
  accept(action: IFormAction): boolean {
    return action.type === FormActionType.LOOP;
  }

  async run(action: LoopFormAction, context: ActionContext, chain: ActionChain): Promise<void> {
    const items = this.getLoopItems(action, context);

    for (let i = 0; i < items.length; i++) {
      context.loopContext.push({ index: i, item: items[i] });
      try {
        const subChain = new ActionChain(action.actions, chain.services);
        await subChain.run(context);
      } catch (e) {
        if (e instanceof BreakException) break;
        if (e instanceof ContinueException) continue;
        throw e;
      } finally {
        context.loopContext.pop();
      }
    }
    await chain.next(context);
  }
}
```
