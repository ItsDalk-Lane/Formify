## 参考代码
```typescript
// src/service/ScriptExecutionService.ts
export class ScriptExecutionService {
  async executeScript(options: ScriptOptions): Promise<any> {
    const { code, context, timeout = 5000 } = options;

    return new Promise((resolve, reject) => {
      const timer = setTimeout(() => reject(new Error('Script timeout')), timeout);

      try {
        const fn = new Function('context', code);
        const result = fn(context);
        clearTimeout(timer);
        resolve(result);
      } catch (error) {
        clearTimeout(timer);
        reject(error);
      }
    });
  }
}
```
