// src/features/tars/providers/types.ts
export interface AIProvider {
  id: string;
  name: string;
  vendor: Vendor;
  chat(messages: Message[], options?: ChatOptions): Promise<string>;
  chatStream(messages: Message[], options?: ChatOptions): AsyncGenerator<string>;
}

export interface ProviderSettings {
  id: string;
  tag: string;
  vendor: Vendor;
  model: string;
  options: {
    apiKey?: string;
    baseUrl?: string;
    temperature?: number;
    maxTokens?: number;
  };
  enabled: boolean;
}
