## 参考代码
```typescript
// src/service/FileOperationService.ts
export class FileOperationService {
  constructor(private app: App) {}

  async writeFile(options: WriteFileOptions): Promise<TFile> {
    const { filePath, content, openAfterCreate } = options;
    const folder = this.app.vault.getAbstractFileByPath(dirname(filePath));
    if (!folder) {
      await this.app.vault.createFolder(dirname(filePath));
    }
    const file = await this.app.vault.create(filePath, content);
    if (openAfterCreate) {
      await this.app.workspace.openLinkText(filePath, '', false);
    }
    return file;
  }
}
```
